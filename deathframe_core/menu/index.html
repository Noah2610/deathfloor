[−][src]Module deathframe_core::menu

States can implement the Menu trait. Provides simple ui helper methods.

Boilerplate


 use deathframe_core::menu::prelude::*;

 type MyGameData = ();
 type MyStateEvent = ();

 #[derive(Default)]
 struct ExampleState {
     ui_data: UiData,
 }

 impl<MyGameData, MyStateEvent> Menu<MyGameData, MyStateEvent> for ExampleState {
     fn event_triggered(
         &mut self,
         data: &mut StateData<MyGameData>,
         event_name: String,
         _event: UiEvent,
     ) -> Option<Trans<MyGameData, MyStateEvent>> {
         match event_name.as_ref() {
             "btn_quit" => Some(Trans::Pop),
             _ => None,
         }
     }

     fn ui_data(&self) -> &UiData {
         &self.ui_data
     }

     fn ui_data_mut(&mut self) -> &mut UiData {
         &mut self.ui_data
     }
 }
